```
import base64

encode_bin=''
with open('log.db','r') as f:
    # print(f.readlines()[0])
    aaa= f.readlines()[0]
    encode_bin = base64.b64decode(aaa)
    print(encode_bin)

with open('result.txt','wb') as f:
    f.write(encode_bin)
```
